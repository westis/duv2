# Description: Create the /api/v1/runners.php endpoint to provide runner profile information.
# Details:
Create a file named 'runners.php' in the /api/v1/ directory:
```php
<?php
require 'helpers.php';
$db = connectToDatabase();
$hasApiKey = validateApiKey($db, false); // optional auth for restricted fields

// Required parameter
$runnerId = $_GET['id'] ?? null;
if (!$runnerId) {
  http_response_code(400);
  outputJson(['error' => 'Runner ID is required']);
}

$query = "SELECT
  ID AS id,
  FirstName AS first_name,
  LastName AS last_name,
  CountryCode AS country";
  
// Add restricted fields if API key is provided
if ($hasApiKey) {
  $query .= ",
  Email AS email,
  DateOfBirth AS date_of_birth,
  Gender AS gender";
}

$query .= "
FROM trunner
WHERE ID=?";

$stmt = $db->prepare($query);
$stmt->bind_param('i', $runnerId);
$stmt->execute();
$result = $stmt->get_result();

if ($result->num_rows === 0) {
  http_response_code(404);
  outputJson(['error' => 'Runner not found']);
}

$runner = $result->fetch_assoc();

// Get runner's performances
$perfQuery = "SELECT
  p.ID AS id,
  e.EventName AS event_name,
  e.ID AS event_id,
  DATE_FORMAT(e.Startdate,'%Y-%m-%d') AS event_date,
  p.Place AS place,
  CONCAT(FLOOR(p.Time/3600), ':', LPAD(FLOOR((p.Time%3600)/60), 2, '0'), ':', LPAD(p.Time%60, 2, '0')) AS time_formatted
FROM tperformance p
JOIN tevent e ON p.EventID = e.ID
WHERE p.RunnerID=?
ORDER BY e.Startdate DESC";

$perfStmt = $db->prepare($perfQuery);
$perfStmt->bind_param('i', $runnerId);
$perfStmt->execute();
$perfResult = $perfStmt->get_result();

$performances = [];
while ($row = $perfResult->fetch_assoc()) {
  $performances[] = $row;
}

$runner['performances'] = $performances;

outputJson($runner);
```
Ensure field names match those defined in the OpenAPI specification.

# Test Strategy:
Test the endpoint with valid and invalid runner IDs. Test with and without API keys to verify restricted fields are only included with valid authentication. Verify the response format matches the OpenAPI specification and includes the runner's performances.
